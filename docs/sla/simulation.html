304        """
305        Continuously plots anomaly detection results and events in real-time from the main thread.
306        This method is designed to run in a loop, updating the plot with data from a queue until
307        the streaming process is deactivated and the queue is empty. It visualizes anomaly scores,
308        thresholds, and events on a time series plot.
309
310        Key Features:
311
312            - Uses Matplotlib to create a real-time plot of anomaly scores.
313
314            - Displays anomaly scores as a line plot.

319
320            - Visualizes events as vertical lines and shaded regions with customizable labels and colors.
321
322        Interactive Mode:
323
324                - Uses Matplotlib's interactive mode (`plt.ion()`) to update the plot in real-time.
325
326                - Ensures the plot is cleared and updated with new data during each iteration.
327
328        Cleanup:
329                - Disables interactive mode (`plt.ioff()`) and closes all figures to free memory after
330                the loop ends.
331
332        Raises:
333            queue.Empty: If the plot queue is empty and no data is available within the timeout.
334
335        Notes:
336            - This method assumes `self.plot_queue` is a thread-safe queue containing tuples of
337              filtered data and anomaly scores.

340              'color', and 'label' to define event visualization.
341        """

352                current_threshold = self._calculate_dynamic_threshold() if self.dynamic_threshold else self.threshold
353                plt.axhline(y=current_threshold, color='cyan', linestyle='--', label='Threshold')
354
355                # Highlight anomalies
356                for i in range(len(filtered_scores)):
357                    if filtered_scores[i] * -1 > current_threshold:
358                        plt.scatter(filtered_data.index[i], filtered_scores[i] * -1, color='red')
359
360                # Plot events as vertical lines
361                if self.events is not None:
362                    for _, event in self.events.iterrows():
363                        event_start = pd.to_datetime(event['start'])
364                        event_end = pd.to_datetime(event['end'])
365                        color = event['color']
366                        label = event['label']
367                        if event_start.date() == filtered_data.index[-1].date():
368                            plt.axvline(x=event_start, color=color, linestyle='--', label=label)
369                            plt.axvspan(event_start, event_end, color=color, alpha=0.3)
370                            plt.axvline(x=event_end, color=color, linestyle='--')
371
372                plt.xlabel('Time')
373                plt.ylabel('Anomaly Score')
374                plt.title('Anomaly Detection with Events')
375                plt.xlim(filtered_data.index[0], filtered_data.index[-1])
376                plt.legend()

384    def process_stream(self):
385        """
386        Consume data from the queue and process it with the anomaly detector.
387
388        This method continuously retrieves data chunks from a queue, processes them,
389        and performs anomaly detection using a pre-trained detector. The results are
390        then prepared for visualization and sent to a plotting queue.
391
392        Workflow:
393
394            1- Retrieve a chunk of data from the queue.
395
396            2- Append the chunk to the historical data, maintaining a rolling window of the last 5000 records.
397
398            3- Preprocess the data for anomaly detection.
399
400            4- Train the anomaly detector if it is not already trained and sufficient data is available.
401
402            5- Perform anomaly detection on the current chunk if the detector is trained.
403
404            6- Update historical scores and count the number of anomalies detected in the current chunk.
405
406            7- Filter historical data and scores to include only the last three hours of data.
407
408            8- Send the filtered data and scores to the plotting queue for visualization.
409
410        Exceptions:
411
412            - Handles `queue.Empty` exceptions when the queue is empty and continues processing.
413
414            - Catches and logs any other exceptions that occur during processing.
415
416        Attributes:
417
418            - self.streaming_active (bool): Flag to control the streaming process.
419
420            - self.queue (queue.Queue): Queue from which data chunks are consumed.
421
422            - self.historical_data (pd.DataFrame): DataFrame storing historical data for processing.
423
424            - self.historical_scores (np.ndarray): Array storing historical anomaly scores.
425
426            - self.window_size (int): Size of the rolling window for preprocessing.
427
428            - self.detector (object): Anomaly detector instance with `fit` and `detect` methods.
429
430            - self.plot_queue (queue.Queue): Queue to send data and scores for visualization.
431        """
432        
433        while self.streaming_active:
434            try:
435                chunk = self.queue.get(timeout=1)
436                if chunk is not None:
437                    self.historical_data = pd.concat([self.historical_data, chunk]).iloc[-5000:]
438                    processed_data = self.preprocess(self.historical_data.iloc[-self.window_size:])
439
440                    if not self.detector.trained and len(processed_data) >= self.detector.window_size:
441                        self.detector.fit(processed_data)
442
443                    if self.detector.trained:
444                        predictions, scores = self.detector.detect(self.preprocess(chunk))
445                        self.historical_scores = np.append(self.historical_scores, scores)
446
447                        anomaly_count = np.sum(predictions == -1)
448                        print(f"Detected {anomaly_count} anomalies in the current chunk.")
449
450                        self.historical_data.index = pd.to_datetime(self.historical_data.index)
451                        three_hours_ago = self.historical_data.index[-1] - pd.Timedelta(hours=3)
452                        filtered_data = self.historical_data[self.historical_data.index >= three_hours_ago]
453
454                        filtered_scores = self.historical_scores[-len(filtered_data):] if len(self.historical_scores) >= len(filtered_data) else np.pad(self.historical_scores, (len(filtered_data) - len(self.historical_scores), 0), 'constant', constant_values=0)
455
456                        self.plot_queue.put((filtered_data, filtered_scores))
457            except queue.Empty:
458                continue
459            except Exception as e:
460                print(f"Error processing stream: {e}")

463        """
464        Executes the streaming anomaly detection simulation.
465
466        This method starts two separate threads:
467
468            1. A streaming thread that simulates or handles incoming data.
469
470            2. A processing thread that processes the streamed data.
471
472        The simulation remains active until interrupted by a KeyboardInterrupt
473        (e.g., pressing Ctrl+C). Upon interruption, the simulation stops gracefully
474        by setting `self.streaming_active` to False and joining both threads.
475        Additionally, this method handles plotting from the main thread.
476        
477        Raises:
478        
479            KeyboardInterrupt: If the simulation is manually stopped by the user.
480        """
481       
482        self.streaming_active = True
483
484        stream_thread = threading.Thread(target=self._stream_data, daemon=True)
485        process_thread = threading.Thread(target=self.process_stream, daemon=True)
486
487        stream_thread.start()
488        process_thread.start()
489
490        try:
491            self._plot_from_main_thread()
492        except KeyboardInterrupt:
493            print("Stopping simulation...")
494            self.streaming_active = False
495
496        stream_thread.join()
497        process_thread.join()
