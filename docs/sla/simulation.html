225        """
226        Streams data from the data source in chunks and places each chunk into a queue.
227        This method iterates over the data source in increments of `chunk_size` and 
228        streams each chunk into a queue for further processing. The streaming process 
229        can be controlled using the `streaming_active` flag, and a delay between 
230        chunks is introduced using `stream_interval`.
231        Yields:
232            None: This method does not return any value but streams data chunks 
233            into the queue.
234        Attributes:
235            data_source (pd.DataFrame): The source of data to be streamed.
236            chunk_size (int): The size of each data chunk to be streamed.
237            queue (queue.Queue): The queue where data chunks are placed.
238            stream_interval (float): The time interval (in seconds) between streaming 
239                consecutive chunks.
240            streaming_active (bool): A flag to control the streaming process. If set 
241                to False, the streaming stops.
242        """
243       
244        for i in range(0, len(self.data_source), self.chunk_size):
245            if not self.streaming_active:
246                break
247            chunk = self.data_source.iloc[i:i + self.chunk_size]
248            self.queue.put(chunk)
249            time.sleep(self.stream_interval)

251    def _plot_from_main_thread(self):
252        """
253        Continuously plots anomaly detection results and events in real-time from the main thread.
254        This method is designed to run in a loop, updating the plot with data from a queue until
255        the streaming process is deactivated and the queue is empty. It visualizes anomaly scores,
256        thresholds, and events on a time series plot.
257        Key Features:
258        - Displays anomaly scores as a line plot.
259        - Highlights anomalies exceeding the threshold with red scatter points.
260        - Dynamically calculates the threshold if enabled, otherwise uses a static threshold.
261        - Visualizes events as vertical lines and shaded regions with customizable labels and colors.
262        Interactive Mode:
263        - Uses Matplotlib's interactive mode (`plt.ion()`) to update the plot in real-time.
264        - Ensures the plot is cleared and updated with new data during each iteration.
265        Cleanup:
266        - Disables interactive mode (`plt.ioff()`) and closes all figures to free memory after
267          the loop ends.
268        Raises:
269            queue.Empty: If the plot queue is empty and no data is available within the timeout.
270        Notes:
271            - This method assumes `self.plot_queue` is a thread-safe queue containing tuples of
272              filtered data and anomaly scores.
273            - The `self.events` attribute should be a DataFrame with columns 'start', 'end',
274              'color', and 'label' to define event visualization.
275        """
276        
277        plt.ion()  # Enable interactive mode
278        while self.streaming_active or not self.plot_queue.empty():
279            try:
280                filtered_data, filtered_scores = self.plot_queue.get(timeout=1)
281                plt.clf()  # Clear the current figure
282                plt.figure(figsize=(12, 6))
283                plt.plot(filtered_data.index, filtered_scores * -1, label='Anomaly Scores', color='blue')
284
285                # Determine the threshold (static or dynamic)
286                current_threshold = self._calculate_dynamic_threshold() if self.dynamic_threshold else self.threshold
287                plt.axhline(y=current_threshold, color='cyan', linestyle='--', label='Threshold')
288
289                # Highlight anomalies
290                for i in range(len(filtered_scores)):
291                    if filtered_scores[i] * -1 > current_threshold:
292                        plt.scatter(filtered_data.index[i], filtered_scores[i] * -1, color='red')
293
294                # Plot events as vertical lines
295                if self.events is not None:
296                    for _, event in self.events.iterrows():
297                        event_start = pd.to_datetime(event['start'])
298                        event_end = pd.to_datetime(event['end'])
299                        color = event['color']
300                        label = event['label']
301                        if event_start.date() == filtered_data.index[-1].date():
302                            plt.axvline(x=event_start, color=color, linestyle='--', label=label)
303                            plt.axvspan(event_start, event_end, color=color, alpha=0.3)
304                            plt.axvline(x=event_end, color=color, linestyle='--')
305
306                plt.xlabel('Time')
307                plt.ylabel('Anomaly Score')
308                plt.title('Anomaly Detection with Events')
309                plt.xlim(filtered_data.index[0], filtered_data.index[-1])
310                plt.legend()
311                plt.pause(0.01)
312            except queue.Empty:
313                continue
314        plt.ioff()  # Disable interactive mode
315        plt.show()
316        plt.close('all')  # Close all figures to free memory

318    def process_stream(self):
319        """
320        Consume data from the queue and process it with the anomaly detector.
321
322        This method continuously retrieves data chunks from a queue, processes them,
323        and performs anomaly detection using a pre-trained detector. The results are
324        then prepared for visualization and sent to a plotting queue.
325
326        Workflow:
327        - Retrieve a chunk of data from the queue.
328        - Append the chunk to the historical data, maintaining a rolling window of the last 5000 records.
329        - Preprocess the data for anomaly detection.
330        - Train the anomaly detector if it is not already trained and sufficient data is available.
331        - Perform anomaly detection on the current chunk if the detector is trained.
332        - Update historical scores and count the number of anomalies detected in the current chunk.
333        - Filter historical data and scores to include only the last three hours of data.
334        - Send the filtered data and scores to the plotting queue for visualization.
335
336        Exceptions:
337        - Handles `queue.Empty` exceptions when the queue is empty and continues processing.
338        - Catches and logs any other exceptions that occur during processing.
339
340        Attributes:
341        - self.streaming_active (bool): Flag to control the streaming process.
342        - self.queue (queue.Queue): Queue from which data chunks are consumed.
343        - self.historical_data (pd.DataFrame): DataFrame storing historical data for processing.
344        - self.historical_scores (np.ndarray): Array storing historical anomaly scores.
345        - self.window_size (int): Size of the rolling window for preprocessing.
346        - self.detector (object): Anomaly detector instance with `fit` and `detect` methods.
347        - self.plot_queue (queue.Queue): Queue to send data and scores for visualization.
348        """
349        
350        while self.streaming_active:
351            try:
352                chunk = self.queue.get(timeout=1)
353                if chunk is not None:
354                    self.historical_data = pd.concat([self.historical_data, chunk]).iloc[-5000:]
355                    processed_data = self.preprocess(self.historical_data.iloc[-self.window_size:])
356
357                    if not self.detector.trained and len(processed_data) >= self.detector.window_size:
358                        self.detector.fit(processed_data)
359
360                    if self.detector.trained:
361                        predictions, scores = self.detector.detect(self.preprocess(chunk))
362                        self.historical_scores = np.append(self.historical_scores, scores)
363
364                        anomaly_count = np.sum(predictions == -1)
365                        print(f"Detected {anomaly_count} anomalies in the current chunk.")
366
367                        self.historical_data.index = pd.to_datetime(self.historical_data.index)
368                        three_hours_ago = self.historical_data.index[-1] - pd.Timedelta(hours=3)
369                        filtered_data = self.historical_data[self.historical_data.index >= three_hours_ago]
370
371                        filtered_scores = self.historical_scores[-len(filtered_data):] if len(self.historical_scores) >= len(filtered_data) else np.pad(self.historical_scores, (len(filtered_data) - len(self.historical_scores), 0), 'constant', constant_values=0)
372
373                        self.plot_queue.put((filtered_data, filtered_scores))
374            except queue.Empty:
375                continue
376            except Exception as e:
377                print(f"Error processing stream: {e}")

379    def run(self):
380        """
381        Executes the streaming anomaly detection simulation.
382        This method starts two separate threads:
383        1. A streaming thread that simulates or handles incoming data.
384        2. A processing thread that processes the streamed data.
385        The simulation remains active until interrupted by a KeyboardInterrupt
386        (e.g., pressing Ctrl+C). Upon interruption, the simulation stops gracefully
387        by setting `self.streaming_active` to False and joining both threads.
388        Additionally, this method handles plotting from the main thread.
389        Raises:
390            KeyboardInterrupt: If the simulation is manually stopped by the user.
391        """
392       
393        self.streaming_active = True
394
395        stream_thread = threading.Thread(target=self._stream_data, daemon=True)
396        process_thread = threading.Thread(target=self.process_stream, daemon=True)
397
398        stream_thread.start()
399        process_thread.start()
400
401        try:
402            self._plot_from_main_thread()
403        except KeyboardInterrupt:
404            print("Stopping simulation...")
405            self.streaming_active = False
406
407        stream_thread.join()
408        process_thread.join()
